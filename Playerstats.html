# Define general legend/guide for all players
playerTheme <- theme(legend.position = "right", legend.key.size = unit(0.4, 
    "cm"), legend.text = element_text(size = rel(0.8)))

# Convert play time to real time hours
playTimeHours <- (playerstats$playOneMinute/20/60/60)

## Actual plot stuff below

ggplot(data = playerstats, aes(x = player, y = deaths)) + geom_bar(colour = "black", 
    width = 0.7, stat = "identity") + xlab("Player") + ylab("Deathcount") + 
    ggtitle("Deaths on Wurstmineberg") + theme(axis.text.x = element_text(angle = 45, 
    hjust = 1))


ggsave("Plots/Deaths.png")

## Saving 7 x 7 in image


ggplot(data = playerstats, aes(x = player, y = (walkOneCm/1e+06))) + geom_bar(colour = "black", 
    width = 0.7, stat = "identity") + xlab("Player") + ylab("Distance (km)") + 
    ggtitle("Distance walked on Wurstmineberg") + theme(axis.text.x = element_text(angle = 45, 
    hjust = 1))


ggsave("Plots/Distance.png")
